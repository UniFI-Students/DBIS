Cancellare una prenotazione:

DELIMITER $$
CREATE PROCEDURE cancel_booking(PC CHAR(16))
BEGIN
DECLARE temp INT;
SELECT COUNT(*) FROM AppoinmentBooking
WHERE PC = PatientPC INTO temp;
IF (temp>0) THEN DELETE FROM AppoinmentBooking WHERE PC = PatientPC;
SELECT 'booking successfully canceled';
ELSE SELECT 'booking not in the system';
END IF;
END $$
DELIMITER ;


trovare il numero di medici divisi per specializazione:

DELIMITER $$

CREATE PROCEDURE specialization(IN sID int, OUT txt char(20))
BEGIN
DECLARE temp INT;
SELECT COUNT(*) INTO temp
from Medic where SpecializationID = sID;
SET txt = CONCAT('there are ', temp,'medics with ', txt,' specialization');
END $$

DELIMITER ;
